# Runtime requirements for the Gaze Tracking Application
# Install with: pip install -r requirements.txt
#
# IMPORTANT FOR WINDOWS: On fresh Windows installs, you may need:
# 1. Microsoft Visual C++ Build Tools (download from Microsoft)
# 2. Run: python -m pip install --upgrade pip setuptools wheel
#    before installing these requirements

# Build tools (required for Python 3.12+ compatibility)
setuptools>=65.0.0     # Provides distutils compatibility (removed in Python 3.12+)
                      # Required for building/installing packages that use distutils
wheel>=0.40.0         # Wheel package format support (ensures pre-built wheels are used when available)

# Core dependencies (required)
pygame>=2.0.0          # UI framework for the 480x800 window
numpy>=1.20.0          # Numerical operations and array handling

# Configuration (recommended)
PyYAML>=5.4.0          # YAML parser for config.yaml file loading
                       # App will work without this but config.yaml won't be loaded

# Machine Learning (optional - only needed when developpement_xg_boost is false)
xgboost>=1.5.0         # XGBoost model for gaze analysis predictions
                       # Required if using real model inference instead of simulation
joblib>=1.0.0          # Model serialization/loading (for XGBoost MultiOutputRegressor)
                       # Required if loading trained models

# Hardware GPIO (optional - only needed when sim_gpio is false)
pyserial>=3.5          # Serial communication for Arduino/microcontroller GPIO backend
                       # Required if using physical hardware buttons/LEDs instead of keyboard simulation
